i : int := 1;
j : int := 2;

theonly {
    i := 4;
    j := i+i;
}

assertEq i 4;
assert (i=4) & j=8;
split;
print_state;
foreach {
    assert (i=4) & j=8;
    split 5;
    print_state;
}

print_state;

cases {
    case match `(?<X>.*).b.a` using [X : string] :
        assertEq X 'abc' msg='bla bla';
        print_state;
    case match ".*b":
        print_state;
    default:
        print_state;
}
